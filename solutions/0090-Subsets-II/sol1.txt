class Solution:
    def subsetsWithDup(self, nums: List[int]) -> List[List[int]]:
        ans = []

        def find_subsets(i, arr):
            if i == len(nums):
                ans.append(sorted(arr[:]))
                return
            
            arr.append(nums[i])
            find_subsets(i + 1, arr)
            arr.pop()
            find_subsets(i + 1, arr)

        find_subsets(0, [])
        res = []
        for sub in ans:
            if sub not in res:
                res.append(sub)
        return res
