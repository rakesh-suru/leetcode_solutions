class Solution:
    def subsetsWithDup(self, nums: List[int]) -> List[List[int]]:
        ans = []

        def find_subsets(i, arr):
            if i == len(nums):
                temp = sorted(arr[:])
                if temp not in ans:
                    ans.append(temp)
                return
            
            arr.append(nums[i])
            find_subsets(i + 1, arr)
            arr.pop()
            find_subsets(i + 1, arr)

        find_subsets(0, [])
        return ans
