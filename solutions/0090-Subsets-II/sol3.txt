class Solution:
    def subsetsWithDup(self, nums: List[int]) -> List[List[int]]:
        ans = []

        def find_subsets(i, arr):
            ans.append(arr[:])
            
            for j in range(i, len(nums)):
                if j != i and nums[j] == nums[j - 1]:
                    continue
                arr.append(nums[j])
                find_subsets(j + 1, arr)
                arr.pop()

        nums = sorted(nums)
        find_subsets(0, [])
        return ans
