class Solution:
    def isSameAfterReversals(self, num: int) -> bool:
        actual = num
        ans = 0

        while num > 0:
            r = num % 10
            ans = ans * 10 + r
            num = num // 10

        num = ans
        ans = 0

        while num > 0:
            r = num % 10
            ans = ans * 10 + r
            num = num // 10

        return actual == ans
